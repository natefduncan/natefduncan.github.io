Offensive and Defensive Efficiency¶

The model predicts points based on Offensive efficiency (OR) and Defensive efficiency (DR), among other things. This code calculates each team's OR and DR for each game of the regular season.

In [21]:
###Formulas for Possessions and Efficiency (found at BasketballReference.com)###

def tot_poss(tm_fgm, tm_fga, tm_fta, tm_or, tm_dr,  
    tm_to, opp_fgm, opp_fga, opp_fta, opp_or, opp_dr, opp_to):
        return (.5)*((tm_fga + .4*tm_fta - 1.07 * (tm_or / (tm_or + opp_dr)) *
        (tm_fga - tm_fgm) + tm_to) + (opp_fga + .4 * opp_fta - 1.07 * (opp_or / 
        (opp_or + tm_dr)) * (opp_fga - opp_fgm) + opp_to))

def off_rating(tm_poss, tm_pts):
    return tm_pts*100/tm_poss
    
def def_rating(opp_poss, opp_pts):
    return opp_pts*100/opp_poss

def game_perf(team, opponent, year):
    
    tm = season_sum[(season_sum["team"] == team) & (season_sum["year"] == year)]
    opp = season_sum[(season_sum["team"] == opponent) & (season_sum["year"] == year)]
    
    intercept = regr.intercept_
    coef = regr.coef_
    
    model_vars = [tm.or_av, tm.dr_av, tm.wins, tm.losses, tm.poss_av, tm.diff_min, 
    tm.diff_max, tm.poss_av, opp.diff_min, opp.diff_max, opp.or_av, opp.dr_av, opp.wins, 
    opp.losses]
    
    model_vars = [round(np.asarray(i)[0], 2) for i in model_vars]
    
    model_vars[0] = np.random.normal(model_vars[0], np.asarray(tm.or_sd)[0], 1)[0]
    model_vars[1] = np.random.normal(model_vars[1], np.asarray(tm.dr_sd)[0], 1)[0]
    model_vars[10] = np.random.normal(model_vars[10], np.asarray(opp.or_sd)[0], 1)[0]
    model_vars[11] = np.random.normal(model_vars[11], np.asarray(opp.dr_sd)[0], 1)[0]
    
    return int(round(regr.predict(np.asarray(model_vars).reshape(1,-1))[0], 2))
